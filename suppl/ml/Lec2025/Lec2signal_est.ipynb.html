import numpy as np
import matplotlib.pyplot as plt
# Produce signal and noise
A = 2.5
theta = np.pi/10
w0 = np.pi/4
L = 100
n = np.r_[0:L] # n = np.ndarray(L)

x = A*np.cos(w0*n+ theta)
noise_std = 0.25
noise = np.random.normal(0,noise_std,L)
y = x + noise
